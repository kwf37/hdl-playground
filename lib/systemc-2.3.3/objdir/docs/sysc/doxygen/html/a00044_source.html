86  sc_scoped_lock( lockable_type& mtx )
87  : m_ref(mtx)
88  , m_active(true)
89  {
90  m_ref.lock();
91  }
92 
93  bool release()
94  {
95  if( m_active )
96  {
97  m_ref.unlock();
98  m_active = false;
99  return true;
100  }

102  }

107  }
