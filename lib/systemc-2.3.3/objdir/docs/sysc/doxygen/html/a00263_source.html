53 template <class T>
54 inline
55 const T
56 sc_abs( const T& a )
57 {
58  // return ( a >= 0 ? a : -a );
59  // the code below is functionaly the same as the code above; the
60  // difference is that the code below works for all arithmetic
61  // SystemC datatypes.
62  T z( a );
63  z = 0;
64  if( a >= z ) {
65  return a;
66  } else {
67  T c( a );
68  c = -a;
69  return c;
70  }
71 }
