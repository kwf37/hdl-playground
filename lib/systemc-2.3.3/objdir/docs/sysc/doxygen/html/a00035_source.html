55 class SC_API sc_host_semaphore : public sc_semaphore_if
56 {
57 #if SC_CPLUSPLUS >= 201103L
58 
59  struct underlying_type
60  {
61  std::mutex mtx;
62  std::condition_variable cond_var;
63  int value;
64  };
65 
66  void do_init(int init) { m_sem.value = init; }
67  void do_wait()
68  {
69  std::unique_lock<std::mutex> lock(m_sem.mtx);
70  while (m_sem.value <= 0) {
71  m_sem.cond_var.wait(lock);
72  }
73  --m_sem.value;
74  }
75  bool do_trywait()
76  {
77  std::unique_lock<std::mutex> lock(m_sem.mtx);
78  if (m_sem.value <= 0)
79  return false;
80  --m_sem.value;
81  return true;
82  }
83  void do_post()
84  {
85  std::unique_lock<std::mutex> lock(m_sem.mtx);
86  ++m_sem.value;
87  m_sem.cond_var.notify_one();
88  }
89  void do_destroy() { /* no-op */ }
90 
91 #elif defined(WIN32) || defined(_WIN32) // use Windows Semaphore
92 
93  typedef HANDLE underlying_type;
94 
95  void do_init(int init)
96  {
97  m_sem = CreateSemaphore( NULL, init, LONG_MAX, NULL );
98  sc_assert( m_sem != NULL );
99  }
100  void do_wait()
101  { WaitForSingleObject( m_sem, INFINITE ); }
102  bool do_trywait()
103  { return ( WaitForSingleObject( m_sem, 0 ) == WAIT_OBJECT_0 ); }
104  void do_post()
105  { ReleaseSemaphore( m_sem, 1, NULL ); }
106  void do_destroy()
107  { CloseHandle( m_sem ); }
108 
109 #else // use POSIX semaphore
110 
111  typedef sem_t underlying_type;
112 
113  void do_init(int init)
114  {
115  int semaphore_initialized = sem_init( &m_sem, 0, init );
116  sc_assert(semaphore_initialized == 0);
117  }
118  void do_wait() { sem_wait( &m_sem ); }
119  bool do_trywait() { return ( sem_trywait( &m_sem ) == 0 ); }
120  void do_post() { sem_post( &m_sem ); }
121  void do_destroy() { sem_destroy( &m_sem ); }
122 
123 #endif // platform-specific implementation
124 
125 public:
126 
127  // constructors and destructor
128 
129  explicit sc_host_semaphore(int init = 0) : m_sem()
130  { do_init(init); }
131  virtual ~sc_host_semaphore()
132  { do_destroy(); }
133 
134  // interface methods
135 
136  // lock (take) the semaphore, block if not available
137  virtual int wait()
138  { do_wait(); return 0; }
139 
140  // lock (take) the semaphore, return -1 if not available
141  virtual int trywait()
142  { return do_trywait() ? 0 : -1; }
143 
144  // unlock (give) the semaphore
145  virtual int post()
146  { do_post(); return 0; }
147 
148  // get the value of the semaphore (not supported)
149  virtual int get_value() const
150  { sc_assert(false && "not supported on sc_host_semaphore"); return -1; }
151 
152 private:
153  underlying_type m_sem;
154 };
